8 Compute the greatest common divisor and least common multipie of two integers. 
a = int(input("Enter first number: "))
b = int(input("Enter second number: "))
gcd = 1
for i in range(1, min(a, b)+1):
    if a % i == 0 and b % i == 0:
        gcd = i
print("Greatest common divisible =", gcd)
lcm = (a * b) // gcd
print("Lowest common multiple =", lcm)
